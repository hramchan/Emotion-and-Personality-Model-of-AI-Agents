;;; Standard Scale of  “Love” an agent has toward either “parent”, “child, or “God”

;;;  1. No feelings or very low average of feelings of love toward agent

;;;  2. Below average feelings of love towards agent

;;;  3. Average feelings of love towards agent

;;;  4. Above average feelings of love towards agent

;;;  5. Intense feelings of love towards agent

(defun parentAgents ()

  (flet ((prompt (string)

           (format t "~&~a: " string)

           (finish-output)

           (read nil 'eof nil)))

    (let ((age  (prompt "What is your age?"))

          (x (prompt "Describe your feelings in terms of love toward your parents using scale (1 to 5) where 5 reveals intense feeling?"))

          (y (prompt "Think about your relationship towards parents and describe intensity, feeling of love using scale (1 to 5) where 5 is highest intensity?"))

          (z (prompt "Describe your feeling in terms of love towards God using scale (1 to 5) where 5 is the intense feeling?"))

          (g (prompt "Reflect adolescent years between ages 14-18, what is relationship towards parents and describe intensity, feeling of love using scale (1 to 5) where 5 is highest intensity?")))

        

      (format t "~&the average of ~a, ~a, ~a, & ~a is:~%~%~a~%"

              x y z g (/(+ x y z g)4.0)))))

;;;       Love-Parental -> action triggers emotion of reproach
